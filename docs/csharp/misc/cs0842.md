## <a name="example"></a><span data-ttu-id="45c48-109">例</span><span class="sxs-lookup"><span data-stu-id="45c48-109">Example</span></span>

namespace TestNamespace  
{  
    [StructLayout(LayoutKind.Explicit)]

        public int Num // CS0842

        {  
            return 1;  
        }  
    }  
}
